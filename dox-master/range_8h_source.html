87  template <typename Index1, typename Index2,
88  typename = std::enable_if_t<detail::is_integral_range_v<Index1> &&
89  detail::is_integral_range_v<Index2>>>
90  void init_range_data(const Index1& lower_bound, const Index2& upper_bound) {
91  // Construct temp pointers
92  auto* MADNESS_RESTRICT const lower = data_;
93  auto* MADNESS_RESTRICT const upper = lower + rank_;
94  auto* MADNESS_RESTRICT const extent = upper + rank_;
95  auto* MADNESS_RESTRICT const stride = extent + rank_;
96 
97  // Set the volume seed
98  volume_ = 1ul;
99  offset_ = 0l;
100 
101  // initialize bounds and extents
102  auto lower_it = std::begin(lower_bound);
103  auto upper_it = std::begin(upper_bound);
104  auto lower_end = std::end(lower_bound);
105  auto upper_end = std::end(upper_bound);
106  for (int d = 0; lower_it != lower_end && upper_it != upper_end;
107  ++lower_it, ++upper_it, ++d) {
108  // Compute data for element d of lower, upper, and extent
109  const auto lower_bound_d = *lower_it;
110  const auto upper_bound_d = *upper_it;
111  lower[d] = lower_bound_d;
112  upper[d] = upper_bound_d;
113  // Check input dimensions
114  TA_ASSERT(lower[d] <= upper[d]);
115  extent[d] = upper[d] - lower[d];
116  TA_ASSERT(extent[d] ==
117  static_cast<index1_type>(upper_bound_d - lower_bound_d));
118  }
119 
120  // Set the volume seed
121  volume_ = 1ul;
122  offset_ = 0l;
123 
124  // Compute strides, volume, and offset, starting with last (least
125  // significant) dimension
126  for (int d = int(rank_) - 1; d >= 0; --d) {
127  stride[d] = volume_;
128  offset_ += lower[d] * stride[d];
129  volume_ *= extent[d];
130  }
131  }

274  void init_range_data(
275  const Permutation& perm,
276  const index1_type* MADNESS_RESTRICT const other_lower_bound,
277  const index1_type* MADNESS_RESTRICT const other_upper_bound) {
278  // Create temporary pointers to this range data
279  auto* MADNESS_RESTRICT const lower = data_;
280  auto* MADNESS_RESTRICT const upper = lower + rank_;
281  auto* MADNESS_RESTRICT const extent = upper + rank_;
282  auto* MADNESS_RESTRICT const stride = extent + rank_;
283 
284  // Copy the permuted lower, upper, and extent into this range.
285  for (unsigned int i = 0u; i < rank_; ++i) {
286  const auto perm_i = perm[i];
287 
288  // Get the lower bound, upper bound, and extent from other for rank i.
289  const auto other_lower_bound_i = other_lower_bound[i];
290  const auto other_upper_bound_i = other_upper_bound[i];
291  const auto other_extent_i = other_upper_bound_i - other_lower_bound_i;
292 
293  // Store the permuted lower bound, upper bound, and extent
294  lower[perm_i] = other_lower_bound_i;
295  upper[perm_i] = other_upper_bound_i;
296  extent[perm_i] = other_extent_i;
297  }
298 
299  // Recompute stride, offset, and volume
300  volume_ = 1ul;
301  offset_ = 0ul;
302  for (int i = int(rank_) - 1; i >= 0; --i) {
303  const auto lower_i = lower[i];
304  const auto extent_i = extent[i];
305  stride[i] = volume_;
306  offset_ += lower_i * volume_;
307  volume_ *= extent_i;
308  }
309  }

822  const auto* MADNESS_RESTRICT const upper = lower + rank_;

836 #endif
