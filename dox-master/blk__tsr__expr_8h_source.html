109 class BlkTsrExprBase : public Expr<Derived> {
110  public:
111  typedef BlkTsrExprBase<Derived> BlkTsrExprBase_;
112  typedef Expr<Derived> Expr_;
113  typedef typename ExprTrait<Derived>::array_type array_type;
115  typedef typename ExprTrait<Derived>::reference reference;

118  protected:

120  std::string vars_;
121  container::svector<std::size_t>
122  lower_bound_;
123  container::svector<std::size_t>
124  upper_bound_;
125 
126  void check_valid() const {
127  const unsigned int rank = array_.trange().tiles_range().rank();
128  // Check the dimension of the lower block bound
129  using std::size;
130  if (size(lower_bound_) != rank) {
131  if (TiledArray::get_default_world().rank() == 0) {
132  TA_USER_ERROR_MESSAGE(
133  "The size lower bound of the block is not equal to rank of "
134  "the array: "
135  << "\n array rank = " << array_.trange().tiles_range().rank()
136  << "\n lower bound = " << lower_bound_);
137 
138  TA_EXCEPTION(
139  "The size lower bound of the block is not equal to "
140  "rank of the array.");
141  }
142  }
143 
144  // Check the dimension of the upper block bound
145  if (size(upper_bound_) != rank) {
146  if (TiledArray::get_default_world().rank() == 0) {
147  TA_USER_ERROR_MESSAGE(
148  "The size upper bound of the block is not equal to rank of "
149  "the array: "
150  << "\n array rank = " << rank
151  << "\n upper bound = " << upper_bound_);
152 
153  TA_EXCEPTION(
154  "The size upper bound of the block is not equal to "
155  "rank of the array.");
156  }
157  }
158 
159  const bool lower_bound_check =
160  std::equal(std::begin(lower_bound_), std::end(lower_bound_),
161  array_.trange().tiles_range().lobound_data(),
162  [](std::size_t l, std::size_t r) { return l >= r; });
163  const bool upper_bound_check =
164  std::equal(std::begin(upper_bound_), std::end(upper_bound_),
165  array_.trange().tiles_range().upbound_data(),
166  [](std::size_t l, std::size_t r) { return l <= r; });
167  if (!(lower_bound_check && upper_bound_check)) {
168  if (TiledArray::get_default_world().rank() == 0) {
169  TA_USER_ERROR_MESSAGE(
170  "The block range is not a sub-block of the array range: "
171  << "\n array range = " << array_.trange().tiles_range()
172  << "\n block range = [ " << lower_bound_ << " , " << upper_bound_
173  << " )");
174  }
175 
176  TA_EXCEPTION("The block range is not a sub-block of the array range.");
177  }
178 
179  const bool lower_upper_bound_check =
180  std::equal(std::begin(lower_bound_), std::end(lower_bound_),
181  std::begin(upper_bound_),
182  [](std::size_t l, std::size_t r) { return l < r; });
183  if (!lower_upper_bound_check) {
184  if (TiledArray::get_default_world().rank() == 0) {
185  TA_USER_ERROR_MESSAGE(
186  "The block lower bound is not less than the upper bound: "
187  << "\n lower bound = " << lower_bound_
188  << "\n upper bound = " << upper_bound_);
189  }
190 
191  TA_EXCEPTION("The block lower bound is not less than the upper bound.");
192  }
193  }
194 
195  public:
196  // Compiler generated functions
197  BlkTsrExprBase(const BlkTsrExprBase_&) = default;
198  BlkTsrExprBase(BlkTsrExprBase_&&) = default;
199  ~BlkTsrExprBase() = default;
200  BlkTsrExprBase_& operator=(const BlkTsrExprBase_&) = delete;
201  BlkTsrExprBase_& operator=(BlkTsrExprBase_&&) = delete;

213  TiledArray::detail::is_integral_range_v<Index1> &&
214  TiledArray::detail::is_integral_range_v<Index2>>>
215  BlkTsrExprBase(reference array, const std::string& vars,
216  const Index1& lower_bound, const Index2& upper_bound)
217  : Expr_(),
218  array_(array),
219  vars_(vars),
220  lower_bound_(std::begin(lower_bound), std::end(lower_bound)),
221  upper_bound_(std::begin(upper_bound), std::end(upper_bound)) {

233  template <typename PairRange,
234  typename = std::enable_if_t<
235  TiledArray::detail::is_gpair_range_v<PairRange>>>
236  BlkTsrExprBase(reference array, const std::string& vars,
237  const PairRange& bounds)
238  : Expr_(), array_(array), vars_(vars) {
239  const auto rank = array.range().rank();
240  lower_bound_.reserve(rank);
241  upper_bound_.reserve(rank);
242  int d = 0;
243  for (auto&& bound_d : bounds) {
244  TA_ASSERT(d < rank);
245  lower_bound_.push_back(TiledArray::detail::at(bound_d, 0));
246  upper_bound_.push_back(TiledArray::detail::at(bound_d, 1));
247  ++d;
248  }
249  TA_ASSERT(d == rank);
250 #ifndef NDEBUG
251  check_valid();
252 #endif // NDEBUG
253  }
