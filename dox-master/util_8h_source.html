37 template <typename T>
38 void scalapack_zero_triangle(
39  blacspp::Triangle tri, scalapack::BlockCyclicMatrix<T>& A, bool zero_diag = false
40 ) {
41 
42  auto zero_el = [&]( size_t I, size_t J ) {
43  if( A.dist().i_own(I,J) ) {
44  auto [i,j] = A.dist().local_indx(I,J);
45  A.local_mat()(i,j) = 0.;
46  }
47  };
48 
49  auto [M,N] = A.dims();
50 
51  // Zero the lower triangle
52  if( tri == blacspp::Triangle::Lower ) {
53 
54  if( zero_diag )
55  for( size_t j = 0; j < N; ++j )
56  for( size_t i = j; i < M; ++i )
57  zero_el( i,j );
58  else
59  for( size_t j = 0; j < N; ++j )
60  for( size_t i = j+1; i < M; ++i )
61  zero_el( i,j );
62 
63  // Zero the upper triangle
64  } else {
65 
66  if( zero_diag )
67  for( size_t j = 0; j < N; ++j )
68  for( size_t i = 0; i <= std::min(j,M); ++i )
69  zero_el( i,j );
70  else
71  for( size_t j = 0; j < N; ++j )
72  for( size_t i = 0; i < std::min(j,M); ++i )
73  zero_el( i,j );
74 
75  }
76 }
77 
78 }
79 }
