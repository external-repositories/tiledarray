567  template <typename Archive,
568  typename std::enable_if<madness::archive::is_output_archive<
569  Archive>::value>::type* = nullptr>
570  void serialize(Archive& ar) const {
571  // Serialize data for empty tile check
572  bool empty = !static_cast<bool>(pimpl_);
573  ar& empty;
574  if (!empty) {
575  // Serialize tile data
576  ar&* pimpl_;
577  }
578  }
579 
580  template <typename Archive,
581  typename std::enable_if<madness::archive::is_input_archive<
582  Archive>::value>::type* = nullptr>
583  void serialize(Archive& ar) {
584  // Check for empty tile
585  bool empty = false;
586  ar& empty;
587 
588  if (!empty) {
589  // Deserialize tile data
590  tensor_type tensor;
591  ar& tensor;
592 
593  // construct a new pimpl
594  pimpl_ = std::make_shared<T>(std::move(tensor));
595  } else {
596  // Set pimpl to an empty tile
597  pimpl_.reset();
598  }
599  }
